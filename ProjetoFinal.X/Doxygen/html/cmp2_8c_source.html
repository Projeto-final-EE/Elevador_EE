61 /* Disable CMP2 interrupt before configuring the CMP2
62 control register. Although a comparator is disabled, an interrupt can be
63 generated by changing the output polarity with the CxPOL bit of the
64 CMxCON0 register,or by switching the comparator on or off with the CxON
65 bit of the CMxCON0 register.*/
66 PIE2bits.C2IE = 0;

68 // C2HYS disabled; C2SP hi_speed; C2ON enabled; C2POL not inverted; C2OE COUT_internal; C2SYNC asynchronous;

71 // C2INTN intFlag_neg; C2INTP no_intFlag; C2PCH FVR; C2NCH CIN1-;

73
74 // Clearing IF flag before enabling the interrupt.
75 PIR2bits.C2IF = 0;
76
