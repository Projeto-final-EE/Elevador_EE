72bool SPI1_Open(spi1_modes_t spi1UniqueConfiguration)
73{
74 if(!SSP1CON1bits.SSPEN)
75 {
76 SSP1STAT = spi1_configuration[spi1UniqueConfiguration].stat;
77 SSP1CON1 = spi1_configuration[spi1UniqueConfiguration].con1;
78 SSP1CON2 = 0x00;
79 SSP1ADD = spi1_configuration[spi1UniqueConfiguration].add;
80 TRISBbits.TRISB4 = spi1_configuration[spi1UniqueConfiguration].operation;

100void SPI1_ExchangeBlock(void *block, size_t blockSize)
101{
102 uint8_t *data = block;
103 while(blockSize--)
104 {
105 SSP1BUF = *data;
